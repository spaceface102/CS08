source_files := $(wildcard *.c)
object_files := $(patsubst %.c, build/%.o, $(source_files))
exec := a.out

.PHONY: clean all

all: $(exec)


$(exec) : $(object_files)
	gcc $^ -o $@

build/%.o : %.c
	@mkdir -p build
	gcc -c $< -o $@

clean:
	rm -r -f ./build

#adding a @ infront of the command silences the command!
#mkdir -p shuts up mkdir in the case that build is already a directory
